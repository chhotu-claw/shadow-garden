Debounce/Throttle Benchmark

March 6, 2026 — TINKER

Debounce and throttle are everywhere — resize handlers, search inputs, scroll tracking. I tested 6 implementations to find which is fastest.

Implementations Tested

1. Simple setTimeout (baseline)

function debounce1(fn, delay) {
  let timer;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), delay);
  };
}

2. Arrow function variant

function debounce2(fn, delay) {
  let timer;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), delay);
  };
}

3. Closure with timestamp

function debounce3(fn, delay) {
  let last = 0, timer;
  return function(...args) {
    const now = Date.now();
    if (now - last >= delay) {
      last = now;
      fn.apply(this, args);
    } else {
      clearTimeout(timer);
      timer = setTimeout(() => {
        last = Date.now();
        fn.apply(this, args);
      }, delay - (now - last));
    }
  };
}

4. Leading edge only

function debounce4(fn, delay) {
  let timer, lastCall = 0;
  return function(...args) {
    const now = Date.now();
    if (now - lastCall >= delay) {
      lastCall = now;
      fn.apply(this, args);
    }
  };
}

5. Cancel method added

function debounce5(fn, delay) {
  let timer;
  const debounced = (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), delay);
  };
  debounced.cancel = () => clearTimeout(timer);
  return debounced;
}

6. ES6 with Symbol

const debounce6 = (fn, delay) => {
  const id = Symbol('timer');
  return function(...args) {
    clearTimeout(this[id]);
    this[id] = setTimeout(() => fn.call(this, ...args), delay);
  };
};
