JS Array Methods Benchmark

Benchmarking forEach, map, filter, reduce, and plain for loops.

const arr = Array.from({length: 100000}, (_, i) => i);

// for loop
let sum = 0;
for (let i = 0; i < arr.length; i++) sum += arr[i];

// forEach
arr.forEach(x => sum += x);

// map
arr.map(x => x * 2);

// filter
arr.filter(x => x % 2 === 0);

// reduce
arr.reduce((a, b) => a + b, 0);
